using System;
using System.Drawing;
using System.Threading;
using System.Windows.Forms;

namespace DrawingApp
{
    public partial class Form1 : Form
    {
        public Form1()
        {
            InitializeComponent();
        }

        // الحدث الخاص بالرسم على النموذج
        private void Form1_Paint(object sender, PaintEventArgs e)
        {
            // تعيين قلم الرسم
            Pen pen = new Pen(Color.FromArgb(255, 0, 0, 0));

            // رسم خط مستقيم من نقطة (20, 10) إلى (300, 100)
            e.Graphics.DrawLine(pen, 20, 10, 300, 100);

            // رسم مستطيل من النقطة (10, 10) بعرض 30 وطول 50
            e.Graphics.DrawRectangle(pen, 10, 10, 30, 50);

            // رسم دائرة أو بيضاوي باستخدام القلم
            e.Graphics.DrawEllipse(pen, 50, 50, 60, 60);

            // تعريف النص المراد رسمه
            string drawString = "Welcome to our class";

            // تعريف الخط المستخدم في الرسم
            Font drawFont = new Font("Arial", 16, FontStyle.Bold);

            // تعريف الفرشاة (اللون الأصفر)
            SolidBrush drawBrush = new SolidBrush(Color.Yellow);

            // النقطة التي سيتم رسم النص عندها
            float x = 150.0F;
            float y = 50.0F;

            // تعريف تنسيق النص (حيث سيتم عرضه عموديًا)
            StringFormat drawFormat = new StringFormat();
            drawFormat.FormatFlags = StringFormatFlags.DirectionVertical;

            // رسم النص على الشاشة
            e.Graphics.DrawString(drawString, drawFont, drawBrush, x, y, drawFormat);

            // رسم دائرة متحركة أفقياً
            Pen pen2 = new Pen(Color.White);
            int i = 0;
            while (i <= 200)
            {
                e.Graphics.Clear(Color.White);  // مسح الشاشة في كل مرة لتحديث الرسم
                e.Graphics.DrawEllipse(pen, i, 100, 30, 30);
                Thread.Sleep(50);  // تأخير قليلاً لإظهار الحركة
                if (i == 200) { break; }
                e.Graphics.DrawEllipse(pen2, i, 100, 30, 30);
                i++;
            }
        }
    }
}
